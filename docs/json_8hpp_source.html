178 void surrogate_pair_to_utf8(uint16_t w1, uint16_t w2, Out &out) {
179 
180  uint32_t cp;
181  if ((w1 & 0xfc00) == 0xd800) {
182  if ((w2 & 0xfc00) == 0xdc00) {
183  cp = 0x10000 + (((static_cast<uint32_t>(w1) & 0x3ff) << 10) | (w2 & 0x3ff));
184  } else {
185  throw invalid_unicode_character();
186  }
187  } else {
188  cp = w1;
189  }
190 
191  if (cp < 0x80) {
192  *out++ = static_cast<uint8_t>(cp);
193  } else if (cp < 0x0800) {
194  *out++ = static_cast<uint8_t>(0xc0 | ((cp >> 6) & 0x1f));
195  *out++ = static_cast<uint8_t>(0x80 | (cp & 0x3f));
196  } else if (cp < 0x10000) {
197  *out++ = static_cast<uint8_t>(0xe0 | ((cp >> 12) & 0x0f));
198  *out++ = static_cast<uint8_t>(0x80 | ((cp >> 6) & 0x3f));
199  *out++ = static_cast<uint8_t>(0x80 | (cp & 0x3f));
200  } else if (cp < 0x1fffff) {
201  *out++ = static_cast<uint8_t>(0xf0 | ((cp >> 18) & 0x07));
202  *out++ = static_cast<uint8_t>(0x80 | ((cp >> 12) & 0x3f));
203  *out++ = static_cast<uint8_t>(0x80 | ((cp >> 6) & 0x3f));
204  *out++ = static_cast<uint8_t>(0x80 | (cp & 0x3f));
205  }

291 
292  if (it != path.end()) {
293 
294  // normal or URI fragment notation?
295  if (*it == '#') {
296  ++it;
297  uri_format = true;
298  }
299 
300  while (it != path.end()) {
301  if (*it++ != '/') {
302  throw invalid_path();
303  }
304 
305  std::string reference_token;
306  while (it != path.end() && *it != '/') {
307  char ch = *it;
308 
309  if (!uri_format) {
310  if (ch == '~') {
311 
312  // ~1 -> /
313  // ~0 -> ~
314 
315  ++it;
316  if (it == path.end()) {
317  throw invalid_reference_escape();
318  }
319 
320  switch (*it) {
321  case '0':
322  ch = '~';
323  break;
324  case '1':
325  ch = '/';
326  break;
327  default:
328  throw invalid_reference_escape();
329  }
330  }
331  } else {
332  // %XX -> char(0xXX)
333 
334  if (ch == '%') {
335  ++it;
336  if (it == path.end()) {
337  throw invalid_reference_escape();
338  }
339 
340  char hex[2];
341  if (!isxdigit(*it)) {
342  throw invalid_reference_escape();
343  }
344 
345  hex[0] = *it++;
346  if (it == path.end()) {
347  throw invalid_reference_escape();
348  }
349 
350  if (!isxdigit(*it)) {
351  throw invalid_reference_escape();
352  }
353 
354  hex[1] = *it;
355 
356  ch = static_cast<char>((detail::to_hex(hex[0]) << 4) | (detail::to_hex(hex[1])));
357  } else if (ch == '~') {
358  // ~1 -> /
359  // ~0 -> ~
360 
361  ++it;
362  if (it == path.end()) {
363  throw invalid_reference_escape();
364  }
365 
366  switch (*it) {
367  case '0':
368  ch = '~';
369  break;
370  case '1':
371  ch = '/';
372  break;
373  default:
374  throw invalid_reference_escape();
375  }
376  }
377  }
378 
379  reference_token.push_back(ch);
380  ++it;
381  }
382

1306 inline std::string escape_string(NS::string_view s, Options options) {
1307 
1308  std::string r;
1309  r.reserve(s.size());
1310 
1311  if (options & Options::EscapeUnicode) {
1312  struct state_t {
1313  unsigned int
1314  expected : 4,
1315  seen : 4,
1316  reserved : 24;
1317  };
1318 
1319  state_t shift_state = {0, 0, 0};
1320  char32_t result = 0;
1321 
1322  for (auto it = s.begin(); it != s.end(); ++it) {
1323 
1324  const auto ch = static_cast<uint8_t>(*it);
1325 
1326  if (shift_state.seen == 0) {
1327 
1328  if ((ch & 0x80) == 0) {
1329  switch (ch) {
1330  case '\"':
1331  r += "\\\"";
1332  break;
1333  case '\\':
1334  r += "\\\\";
1335  break;
1336 #if 0
1337  case '/': r += "\\/"; break;
1338 #endif
1339  case '\b':
1340  r += "\\b";
1341  break;
1342  case '\f':
1343  r += "\\f";
1344  break;
1345  case '\n':
1346  r += "\\n";
1347  break;
1348  case '\r':
1349  r += "\\r";
1350  break;
1351  case '\t':
1352  r += "\\t";
1353  break;
1354  default:
1355  if (!isprint(ch)) {
1356  r += "\\u";
1357  char buf[5];
1358  snprintf(buf, sizeof(buf), "%04X", ch);
1359  r += buf;
1360  } else {
1361  r += static_cast<char>(ch);
1362  }
1363  break;
1364  }
1365  } else if ((ch & 0xe0) == 0xc0) {
1366  // 2 byte
1367  result = ch & 0x1f;
1368  shift_state.expected = 2;
1369  shift_state.seen = 1;
1370  } else if ((ch & 0xf0) == 0xe0) {
1371  // 3 byte
1372  result = ch & 0x0f;
1373  shift_state.expected = 3;
1374  shift_state.seen = 1;
1375  } else if ((ch & 0xf8) == 0xf0) {
1376  // 4 byte
1377  result = ch & 0x07;
1378  shift_state.expected = 4;
1379  shift_state.seen = 1;
1380  } else if ((ch & 0xfc) == 0xf8) {
1381  // 5 byte
1382  throw invalid_utf8_string(); // Restricted by RFC 3629
1383  } else if ((ch & 0xfe) == 0xfc) {
1384  // 6 byte
1385  throw invalid_utf8_string(); // Restricted by RFC 3629
1386  } else {
1387  throw invalid_utf8_string(); // should never happen
1388  }
1389  } else if (shift_state.seen < shift_state.expected) {
1390  if ((ch & 0xc0) == 0x80) {
1391  result <<= 6;
1392  result |= ch & 0x3f;
1393  // increment the shift state
1394  ++shift_state.seen;
1395 
1396  if (shift_state.seen == shift_state.expected) {
1397  // done with this character
1398 
1399  char buf[5];
1400 
1401  if (result < 0xd800 || (result >= 0xe000 && result < 0x10000)) {
1402  r += "\\u";

1404  r += buf;
1405  } else {
1406  result = (result - 0x10000);
1407 
1408  r += "\\u";
1409  snprintf(buf, sizeof(buf), "%04X", 0xd800 + ((result >> 10) & 0x3ff));
1410  r += buf;
1411 
1412  r += "\\u";
1413  snprintf(buf, sizeof(buf), "%04X", 0xdc00 + (result & 0x3ff));
1414  r += buf;
1415  }
1416 
1417  shift_state.seen = 0;
1418  shift_state.expected = 0;
1419  result = 0;
1420  }
1421 
1422  } else {
1423  throw invalid_utf8_string(); // should never happen
1424  }
1425  } else {
1426  throw invalid_utf8_string(); // should never happen
1427  }
1428  }
1429  } else {
1430 
1431  for (char ch : s) {
1432 
1433  switch (ch) {
1434  case '\"':
1435  r += "\\\"";

1437  case '\\':
